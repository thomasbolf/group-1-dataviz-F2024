
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the data
data = pd.read_csv('/mnt/data/file-SUEtyrjUwn57IhwoCDvwfeCF')

# Convert the numerical columns to appropriate types
num_columns = ["SepalLengthCm", "SepalWidthCm", "PetalLengthCm", "PetalWidthCm"]
data[num_columns] = data[num_columns].apply(pd.to_numeric, errors='coerce')

# Set a playful style
sns.set(style="whitegrid", palette="bright")

# Create the pair plot with histograms on the diagonal
pair_plot = sns.pairplot(data, hue='Species', height=2.5, diag_kind='hist',
                         plot_kws={'alpha': 0.6, 's': 80, 'edgecolor': 'w'})

# Enhance aesthetics with a fun title and layout adjustments
pair_plot.fig.suptitle("Playful Pair Plot Analysis of Iris Dataset", va='bottom', fontsize=14, color='purple')
plt.subplots_adjust(top=0.9)

# Save the visualization as a PNG file
pair_plot.savefig('/mnt/data/iris_pairplot_histogram.png')
