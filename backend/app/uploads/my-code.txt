import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd

# Load the data
data = pd.read_csv('<file_path>')

# Set style for the plot
sns.set(style="whitegrid")

# Initialize the matplotlib figure
plt.figure(figsize=(10, 6))

# Create scatterplot
sns.scatterplot(data=data, x='bill_length_mm', y='bill_depth_mm',
                hue='species', style='sex',
                palette='muted', s=100)

# Add titles and labels
plt.title('Penguin Bill Dimensions by Species and Sex', fontsize=16)
plt.xlabel('Bill Length (mm)', fontsize=14)
plt.ylabel('Bill Depth (mm)', fontsize=14)

# Customizations for professionalism
plt.legend(title='Species and Sex', title_fontsize='13', loc='upper right')
plt.grid(True, linestyle='--', alpha=0.6)
plt.tight_layout()

# Save the plot as a PNG file
plt.savefig('penguin_bill_dimensions.png')
